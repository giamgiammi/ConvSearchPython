Module convSearchPython.pipelines.filtered

This module contains a collection of pipelines that apply a reranking filtering that remove (or penalize) relevant docs found for an utterance from the following utterances' results (inside a conversation).

The filter work is guided by 2 parameter:

  • multiplier: this is the multiplicative factor used to reduce the score of the already seen documents. 0 \le multiplier < 1
  • max_rank: this indicates the maximum rank to consider when keeping seen docs in memory. Given that a normal search pull 1000 documents, it's useless to consider all the seen documents. Instead, we want to focus on the top-k documents that our system considered relevant

The filter is implemented in ranking.seen_docs_filter

"""
This module contains a collection of pipelines that apply
a reranking filtering that remove (or penalize) relevant docs found for
an utterance from the following utterances' results (inside
a conversation).

The filter work is guided by 2 parameter:

- multiplier: this is the multiplicative factor used to reduce the score of the
              already seen documents. $$ 0 \le multiplier < 1 $$
- max_rank: this indicates the maximum rank to consider when keeping seen docs in memory.
            Given that a normal search pull 1000 documents, it's useless to consider
            **all** the seen documents. Instead, we want to focus on the top-k documents
            that our system considered relevant

The filter is implemented in `ranking.seen_docs_filter`
"""
from multiprocessing import Pool
from typing import Dict, Tuple, List, Optional, Type

import pandas as pd

from convSearchPython.basics import pyterrier as pt
from convSearchPython.pipelines import AbstractPipeline, QueriesStructNeedingInterface, CachedPipelineInterface, \
    AbstractWrapperPipelineInterface
from convSearchPython.ranking.seen_docs_filter import create_filter_seen_docs_filter
from convSearchPython.utils.data_utils import queries_conversation_splitter
from convSearchPython.utils.imports import import_pipeline
from convSearchPython.utils.parallel_utils import RunPipeline


class FilteredPipeline(AbstractWrapperPipelineInterface):
    """
    This pipeline applies docs filtering to an existing pipeline
    """
    def __init__(self, queries: pd.DataFrame, conversations: Dict[str, List[str]],
                 query_map: Dict[str, Tuple[str, int]], pipeline: str, multiplier=0.0, max_rank=20, **kwargs):
        """
        Args:
            queries: queries DataFrame
            conversations: conversation map
            query_map: queries map
            pipeline: the reference name of the pipeline to wrap
            multiplier: seen doc multiplier (to reduce already seen docs relevance)
            max_rank: maximum rank to consider
            **kwargs: all others arguments are passed to the wrapped pipeline
        """
        self._query_map = query_map
        self._conversations = conversations
        cls: Type[AbstractPipeline] = import_pipeline(pipeline)
        if issubclass(cls, QueriesStructNeedingInterface):
            kwargs['queries'] = queries
            kwargs['conversations'] = conversations
            kwargs['query_map'] = query_map
        if not issubclass(cls, CachedPipelineInterface):
            kwargs.pop('cache_dir')
        self._wrapped_pipeline = cls(**kwargs)
        self._multiplier = multiplier
        self._max_rank = max_rank

    @property
    def reranker(self):
        return pt.apply.doc_score(create_filter_seen_docs_filter(self._multiplier, self._max_rank, self._query_map))

    def run_on(self, queries: pd.DataFrame, parallel_pool: Optional[Pool]) -> pd.DataFrame:
        res = self._wrapped_pipeline.run_on(queries, parallel_pool)

        if parallel_pool is None:
            return self.reranker(res)

        it = queries_conversation_splitter(res, self._conversations)
        results = list(parallel_pool.imap(RunPipeline(self, 'reranker'), it))
        return pd.concat(results)

    @property
    def name(self):
        """Name is composed by """
        return self._wrapped_pipeline.name + f'_seenFilter-{self._multiplier}-{self._max_rank}'
