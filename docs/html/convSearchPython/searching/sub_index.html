"""Creating a memory-only index from a run result"""
import logging
from os import environ
from pathlib import Path
from typing import Union, Callable, Tuple, List, Optional

import pandas as pd
import pyterrier as pt
import pyterrier.index as pt_index
import pyterrier.transformer as pt_tr

from convSearchPython.pipelines import RetrieveMetadata, AbstractPipeline
from convSearchPython.utils import random_string, recursive_rm, mkdir

sub_index_dir_info = [False]


def create_sub_index(results: pd.DataFrame, properties: dict, metadata: list) -> Tuple[pt.IndexFactory, Optional[Path]]:
    sub_index_dir = environ.get('SUB_INDEX_DIR')
    if sub_index_dir is None:
        indexer = pt.DFIndexer('', type=pt_index.IndexingType.MEMORY)
        for k, v in properties.items():
            indexer.setProperty(k, v)
        index_ref = indexer.index(results['text'], results[metadata])
        return pt.IndexFactory.of(index_ref), None
    else:
        if not sub_index_dir_info[0]:
            logging.getLogger(__name__).info(f'Using {sub_index_dir} to store temp indexes')
            sub_index_dir_info[0] = True
        dir = Path(sub_index_dir)
        if not (dir.exists() and dir.is_dir()):
            raise Exception(f'invalid SUB_INDEX_DIR {sub_index_dir}')
        dir = Path(dir, random_string(10))
        mkdir(dir)
        indexer = pt.IterDictIndexer(str(dir))
        for k, v in properties.items():
            indexer.setProperty(k, v)
        index_ref = indexer.index(({'docno': d['docno'], 'text': d['text']} for _, d in results.iterrows()),
                                  meta=['docno', 'text'],
                                  meta_lengths=[44, 4096],
                                  fields=['text']
                                  )
        return pt.IndexFactory.of(index_ref), dir


def sub_index_search(
        conv_queries: pd.DataFrame,
        base_transformer_factory: Callable[[int, List[str]], pt_tr.TransformerBase],
        sub_transformer_factory: Callable[[pt.IndexFactory], Union[pt_tr.TransformerBase, AbstractPipeline]],
        index_properties: dict,
        sub_index_size: int,
        num_docs: int,
        metadata: RetrieveMetadata = RetrieveMetadata.ALL,
        return_tuple_of_ds: bool = False,
        rerun_first_query: bool = False
) -> Union[pd.DataFrame, Tuple[pd.DataFrame, ...]]:
    """
    Use a temp sub-index from the first query to run subsequent ones
    Args:
        conv_queries: dataset with same-conversation queries
        base_transformer_factory: callable to create the pyterrier transformer for first query search. Must accept
        the number of docs to retrieve and metadata (list) as arguments. Note: the result of this will be the
        documents in the sub-index
        sub_transformer_factory: callable to create the sub transformer, must accept a pyterrier index.
        Instead of a transformer can return an instance of AbstractPipeline.
        and metadata (list) as parameters
        index_properties: dict with indexing properties
        sub_index_size: number of documents for the sub-index
        num_docs: number of docs to return (only needed for the first query)
        metadata: metadata to retrieve (as RetrieveMetadata obj)
        return_tuple_of_ds: return a tuple of dataset instead of a dataset, useful to avoid pd.concat when
        it will be executed later anyway
        rerun_first_query: re-run the first query on the sub-index
    Returns:
        results dataframe or a tuple of dataframes if "return_tuple_of_ds" is set to True"""

    base_transformer = base_transformer_factory(sub_index_size, RetrieveMetadata.ALL.value)
    first_query_results = base_transformer(conv_queries.iloc[0:1])
    index, dir = create_sub_index(first_query_results, index_properties, metadata.value)
    sub_transformer = sub_transformer_factory(index)

    if rerun_first_query:
        res = sub_transformer(conv_queries)
    else:
        res = sub_transformer(conv_queries[conv_queries['qid'] != conv_queries.iloc[0]['qid']])

    if dir is not None:
        recursive_rm(dir)

    if metadata == RetrieveMetadata.DEFAULT:
        first_query_results.drop('text', 1, inplace=True)

    if rerun_first_query:
        if return_tuple_of_ds:
            return res,
        return res
    if return_tuple_of_ds:
        return first_query_results.iloc[:num_docs], res
    return pd.concat((first_query_results.iloc[:num_docs], res))
